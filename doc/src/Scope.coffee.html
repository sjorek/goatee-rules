ScriptScope = require 'goatee-script.js/lib/Scope' Expressions = require './Expressions'

Scope

class Scope and namespace GoateeRules

class Scope extends ScriptScope

Method create

Create a new Expression or Expressions instance

Parameters:

  • operator must be a String.

  • parameters must be an Array.

Returns an Expressions or an Expression

create : (operator, parameters) -> return @addRule new Expressions, parameters if operator is 'rules' super(operator, parameters)

Method addRule

Add a Expression to the given ExpressionMap rule

Parameters:

  • rule must be an ExpressionMap.

  • parameters must be an Array.

addRule: (rule, parameters) -> [key,expression,important] = parameters if expression.operator.name is 'list' expression = @create 'group', [expression] rule.add key, expression, important module.exports = Scope
